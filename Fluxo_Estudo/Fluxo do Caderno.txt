/* ==============================================================
   DIA 004 — Bits e Bytes
   Objetivo: Entender como os dados são representados e armazenados.
   ==============================================================
   CONCEITOS FUNDAMENTAIS:
   - Bit = menor unidade de informação (0 ou 1)
   - Byte = conjunto de 8 bits
   - Arquivos binários = cabeçalho + dados
   - Cabeçalho = metadados (tipo, tamanho, etc.)
   - Endianness define a ordem dos bytes

   EXEMPLO PRÁTICO:
   -------------------------------------------------------------- */

#include <stdio.h>

int main() {
    FILE *arquivo = fopen("dados.bin", "wb");
    int numeros[] = {1, 2, 3, 4};
    fwrite(numeros, sizeof(int), 4, arquivo);
    fclose(arquivo);

    printf("Arquivo binário criado com sucesso!\n");
    return 0;
}
